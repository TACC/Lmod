    local str="$1";
    local RED='\\033[1;31m';

    echo "${RED}${str}${NONE}"
}
set-function fooFunc {
    echo "arg1: $1"
}

    if [ -n "${ZSH_VERSION:-}" ]; then
        \echo "junk";
    fi

    echo "do not forget \"foo\""

    for arg in "$@";
    do
        printf "%s\n" "$arg" | sed -e "s/'/'\\\\''/g" -e "1s/^/'/" -e "\$s/\$/' \\\\/";
    done;
    echo " "
}
complete bash XyZ123 {-F _xyz123}
